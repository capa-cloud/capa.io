快速开始

简洁、直觉、强大。

// 构建 RPC 客户端
CapaRpcClient client = new CapaRpcClientBuilder()
    .build();

// 调用远程服务 — 自动适配云平台
Mono<String> result = client.invokeMethod(
    "service-app-id",
    "hello",
    "request",
    HttpExtension.POST,
    null,
    TypeRef.STRING
);

// 获取结果
String response = result.block();
